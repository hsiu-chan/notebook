    id: daily_solar_energy_production

    on_value:

            id: daily_solar_production_watts
            text:
              format: "%.1f"
              args: [ 'id(daily_solar_energy_production).state' ]

  - platform: homeassistant
    id: daily_consumption
    entity_id: sensor.envoy_122030047695_energy_consumption_today
    internal: true
    on_value:
      then:
        - lvgl.label.update:
            id: daily_consumption_watts
            text:
              format: "%.1f"
              args: [ 'id(daily_consumption).state' ]

  - platform: combination
    id: daily_net_sum
    type: linear
    name: "Daily net sum"
    internal: true
    sources:
      - source: daily_net_import
        coeffecient: 1.0
      - source: daily_net_export
        coeffecient: -1.0
    on_value:
      then:
        - lvgl.label.update:
            id: daily_net_watts
            text:
              format: "%.1f"
              args: [ 'id(daily_net_sum).state' ]
        - if:
            condition:
              lambda: return (id(daily_net_sum).state < 0);
            then:
              - lvgl.widget.update:
                  id: daily_net_watts
                  text_color: ep_blue
              - lvgl.image.update:
                  id: daily_net_watts_icon
                  src: up_arrow
            else:
              - lvgl.widget.update:
                  id: daily_net_watts
                  text_color: ep_orange
              - lvgl.image.update:
                  id: daily_net_watts_icon
                  src: down_arrow
